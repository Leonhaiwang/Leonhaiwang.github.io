堆排序

2021-03-29

堆排序是一种类似于完全二叉树（子节点总是大于或者小于父节点）的一种数据结构，

平均时间复杂的O(nlogn)

  • 大顶堆：每个节点的值都大于或者等于子节点的值，升序
  • 小顶堆：每个节点的值都小于或者等于子节点的值，降序

排序过程：

  • 将所有的数据存储在堆中
  • 按大顶堆构建堆，将数据从大到小取出，构造出大顶堆
  • 入堆完成后，将堆顶元素取出，末尾元素置于堆顶，调整结构
  • 反复执行
  var len;

  function buildMaxHeap(arr) { //大顶堆
    len = arr.length;
    for (var i = Math.floor(len / 2); i >= 0; i--) {
      heapify(arr, i)
    }
  }
  //堆调整
  function heapify(arr, i) {
    var left = 2 * i,
      right = 2 * i + 2,
      largest = i;
    if (left < len && arr[left] > arr[largest]) {
      largest = left;
    }
    if (right < len && arr[right] > arr[largest]) {
      largest = right;
    }
    if (largest != i) {
      swap(arr, i, largest)
      heapify(arr, largest)
    }
  }

  function swap(arr, i, j) {
    var temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
  }

  function heapSort(arr) {
    buildMaxHeap(arr);
    for (let i = arr.length - 1; i > 0; i--) {
      swap(arr, 0, i);
      len--;
      heapify(arr, 0)
    }
    return arr;
  }
  console.log(heapSort([1, 3, 7, 3, 24, 5]));
